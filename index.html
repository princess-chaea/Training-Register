if (name.length > 50) name = name.substring(0, 50); return name; } /** * 서명 이미지를 드라이브의 지정된 폴더에 저장 */ function saveSignatureToDrive(trainingId, trainingTitle, name, base64Data) { const rootFolder = DriveApp.getFolderById(SIGNATURE_ROOT_ID); const subFolderName = `${trainingTitle}(서명)`; let subFolder; const subs = rootFolder.getFoldersByName(subFolderName); if (subs.hasNext()) { subFolder = subs.next(); } else { subFolder = rootFolder.createFolder(subFolderName); } // 기존 동일인 서명 파일이 있으면 제거 (덮어쓰기 효과) try { const existingFiles = subFolder.getFilesByName(`${name}.png`); while (existingFiles.hasNext()) { existingFiles.next().setTrashed(true); } } catch (e) { console.warn("기존 파일 삭제 중 오류 (무시됨):", e); } function getSheetData(sheetName) { const sheet = ACTIVE_SS.getSheetByName(sheetName); if (!sheet) return []; const headerMap = getHeaderIndices(sheet); const values = sheet.getDataRange().getValues(); if (values.length <= 1) return []; const dataRows = values.slice(1); const keys = Object.keys(headerMap); return dataRows.map(row => { let obj = {}; keys.forEach(key => { const colIdx = headerMap[key]; } obj[key] = val; }); return obj; }); } function getHeaderIndices(sheet) { const lastCol = sheet.getLastColumn(); if (lastCol === 0) return {}; const headers = sheet.getRange(1, 1, 1, lastCol).getValues()[0]; else if (h === 'id' || h === '아이디') key = 'id'; else if (h === 'createdat' || h === '작성일') key = 'createdAt'; else if (h === 'timestamp' || h === '등록시간') key = 'timestamp'; // 등록부에서는 timestamp 사용 else if (h === 'trainingid' || h === '연수id') key = 'trainingId'; else if (h === 'name' || h === '성명' || h === '이름') key = 'name'; else if (h === 'position' || h === '직위' || h === '담당' || h === '담당(직위)') key = 'position'; else if (h === 'grade' || h === '직급' || h === '소속' || h === '근무지' || h === '소속(근무지)') key = 'grade'; else if (h === 'status' || h === '상태') key = 'status'; else if (h === 'content' || h === '내용' || h === '서명' || h === '서명/내용') key = 'content'; if (key) map[key] = i + 1; }); return map; } function createResponse(data) { return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON);
